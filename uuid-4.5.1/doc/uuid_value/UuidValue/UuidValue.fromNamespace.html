const UuidValue.fromNamespace(Namespace ns)
    : uuid = ns == Namespace.nil
          ? InternalConstants.zNIL
          : ns == Namespace.dns
              ? InternalConstants.zDNS
              : ns == Namespace.url
                  ? InternalConstants.zURL
                  : ns == Namespace.oid
                      ? InternalConstants.zOID
                      : ns == Namespace.x500
                          ? InternalConstants.zX500
                          : ns == Namespace.max
                              ? InternalConstants.zMAX
                              : InternalConstants.zNIL;
